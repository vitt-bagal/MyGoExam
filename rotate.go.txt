//Given an array, rotate the array by one position in clock-wise direction.
 
/Example 1:

//Input:
//N = 5
//A[] = {1, 2, 3, 4, 5}
//Output:
//5 1 2 3 4

package main

import (
	"fmt"
)

func main() {
	s := []int{9, 8, 7, 6, 4, 2, 1, 3}
	fmt.Println("Slice before rotate:", s)
	rotate(s)
}

func rotate(a []int) {
	for i := 0; i < len(a)-1; i++ {
		a[i], a[i+1] = a[i+1], a[i]
	}
	fmt.Println("Slice after rotate:", a)
}
